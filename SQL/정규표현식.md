**정규 표현식**

1. ‘+’
    1. 1 ~ 무한
2. ‘?’
    1. 0, 1
3. ‘*’
    1. 0 ~ 무한
4. ‘\’
    1. ESCAPE
5. ‘^’ : 시작 문자
6. ‘[^]’: not

```sql
SELECT REGEXP_SUBSTR('ASDFASDFASDFSDF','.')
FROM DUAL
=> A

SELECT REGEXP_SUBSTR('ASDFASDFASDFSDF','.+')
FROM DUAL
=> ASDFASDFASDFSDF

SELECT REGEXP_SUBSTR('ASDFASDFASDFSDF','.+')
FROM DUAL
=> ASDFASDFASDFSDF

SELECT REGEXP_SUBSTR('ASDFASDF123123123ASDFSDF','.\d+')
FROM DUAL
=> F123123123

SELECT REGEXP_SUBSTR('ASDFASDF123123123ASDFSDF','^\d')
FROM DUAL
=> 

SELECT REGEXP_SUBSTR('ASDFASDF123123123ASDFSDF','^\d?.')
FROM DUAL
=> A

SELECT REGEXP_SUBSTR('ASDFASDF123123123ASDFSDF','\d{3}')
FROM DUAL
=> 123

SELECT REGEXP_SUBSTR('ASDFASDF123123123ASDFSDF','\d{2,}')
FROM DUAL
=> 123123123

SELECT REGEXP_SUBSTR('ASDFASDF123123123ASDFSDF','[123]+')
FROM DUAL
=> 123123123

SELECT REGEXP_SUBSTR('ASDFASDF1230456ASDFSDF','[123]+')
FROM DUAL
=> 123

SELECT REGEXP_SUBSTR('ASDFASDF110493ASDFSDF','[123]+',1,2)
FROM DUAL
=> 3

SELECT REGEXP_SUBSTR('ASDFASDF1104934ASDFSDF','(12)|(34)')
FROM DUAL
=> 34

SELECT REGEXP_SUBSTR('ASD.04934ASDFSDF12312312','\.')
FROM DUAL
=> .

SELECT REGEXP_SUBSTR('ASD.049  대한민국 34ASDFSDF1     2312312','[가-힝]+')
FROM DUAL
=> 대한민국
```

```sql
-- 주민번호 검증
SELECT
	DECODE(SUBSTR(:num,-1)
		,SUBSTR(11-MOD(SUM(SUBSTR(REPLACE(:num,'-'),ROWNUM,1) * SUBSTR('234567892345',ROWNUM,1)),11),-1)
		,'OK','NOT OK') CMP
	,SUBSTR(11-MOD(SUM(SUBSTR(REPLACE(:num,'-'),ROWNUM,1) * SUBSTR('234567892345',ROWNUM,1)),11),-1) CALC
	,SUBSTR(:num,-1) VRF
FROM DUAL
CONNECT BY LEVEL <= LENGTH(REPLACE(:num,'-'))
```
